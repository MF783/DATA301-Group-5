Number of Fatalities per Year

crash_data %>%
  group_by(crashYear) %>%
  summarise(Fatalities = sum(fatalCount)) %>%
  ggplot(aes(x = crashYear, y = Fatalities)) +
  geom_line(color = "red") +
  theme_minimal() +
  labs(x = "Year", y = "Number of Fatalities", title = "Number of Fatalities per Year")
